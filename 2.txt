SELECT miejscowosc.nazwa, miejscowosc.kraj, pomiary.temperatura
FROM miejscowosc

zadanie 4

SELECT miesiace.nazwa, AVG(pomiary.temperatura)
FROM miesiace
JOIN pomiary ON miesiace.id = pomiary.id_miesiac
GROUP BY miesiace.nazwa
